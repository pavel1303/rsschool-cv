                function decToHex(num){
                  let res = [];
                  if(num <= 0){
                    return '00'
                  } else if(num >= 255){
                    return 'FF'
                  } else {
                    let alphabet = '0123456789ABCDEF';
                    while(num){
                    res.unshift(alphabet[num % 16]);
                    num = Math.floor(num / 16);
                  }
                  }
                  return res.join('').length === 2 ? res.join('') : `0${res.join('')}`;
                }
